} } function getLogSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) sheet = ss.insertSheet(SHEET_NAME); if (sheet.getLastRow() === 0) { sheet.appendRow(BASE_HEADERS);
